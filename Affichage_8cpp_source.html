225 * @brief Dessine une route segmentée entre deux points de la scène.
226 *
227 * Chaque segment est un rectangle orienté selon l'angle de la route,
228 * espacés régulièrement et centrés entre les deux villes.
229 * Pour une route double, une voie parallèle décalée de 17 px est ajoutée.
230 * Si @p proprio est non vide, le nom du propriétaire est affiché au centre.
231 *
232 * @complexity O(longueur)
233 *
234 * @param x1 Coordonnée X de la ville de départ (pixels scène).
235 * @param y1 Coordonnée Y de la ville de départ (pixels scène).
236 * @param x2 Coordonnée X de la ville d'arrivée (pixels scène).
237 * @param y2 Coordonnée Y de la ville d'arrivée (pixels scène).
238 * @param couleur Couleur de remplissage des segments de la voie principale.
239 * @param longueur Nombre de segments (= nombre de wagons requis).
240 * @param couleur2 Couleur de la voie parallèle ; Qt::transparent = route simple.
241 * @param proprio Prénom du joueur propriétaire ; vide = route non revendiquée.
242 */
243void Affichage::dessinerRoute(int x1, int y1, int x2, int y2, QColor couleur, int longueur, QColor couleur2, QString proprio)
244{
245 float totalDx = x2 - x1;
246 float totalDy = y2 - y1;
247 float angle = atan2(totalDy, totalDx);
248 float dist = sqrt(totalDx*totalDx + totalDy*totalDy);
249
250 float segmentW = 60.0f;
251 float segmentH = 12.0f;
252 float gap = 6.0f;
253 float total = longueur * segmentW + (longueur - 1) * gap;
254 float rayonVille = 28.0f;
255 float offset = (dist - total) / 2.0f;
256
257 float px = -sin(angle) * 17;
258 float py = cos(angle) * 17;
259
260 for (int i = 0; i < longueur; i++)
261 {
262 float d = offset + i * (segmentW + gap);
263 float x = x1 + cos(angle) * d;
264 float y = y1 + sin(angle) * d;
265
266 QGraphicsRectItem* segment = scene->addRect(0, -segmentH/2, segmentW, segmentH);
267 segment->setBrush(QBrush(couleur));
268 segment->setPen(QPen(Qt::black));
269 QTransform transform;
270 transform.translate(x, y);
271 transform.rotateRadians(angle);
272 segment->setTransform(transform);
273
274 if (couleur2 != Qt::transparent)
275 {
276 QGraphicsRectItem* segment2 = scene->addRect(0, -segmentH/2, segmentW, segmentH);
277 segment2->setBrush(QBrush(couleur2));
278 segment2->setPen(QPen(Qt::black));
279 QTransform transform2;
280 transform2.translate(x + px, y + py);
281 transform2.rotateRadians(angle);
282 segment2->setTransform(transform2);
283 }
284 }
285
286 if (!proprio.isEmpty())
287 {
288 float midX = (x1 + x2) / 2.0f;
289 float midY = (y1 + y2) / 2.0f;
290
291 QGraphicsTextItem* texte = scene->addText(proprio);
292 QFont font;
293 font.setPointSize(8);
294 font.setBold(true);
295 texte->setFont(font);
296 texte->setDefaultTextColor(Qt::darkBlue);
297 texte->setPos(midX - texte->boundingRect().width()/2,
298 midY - texte->boundingRect().height()/2);
299 }
