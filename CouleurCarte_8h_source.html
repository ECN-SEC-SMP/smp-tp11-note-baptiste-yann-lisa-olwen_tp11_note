11/**

16 *
17 * @ingroup Cartes
18 */
19enum class CouleurCarte
20{
21 Bleu, ///< Carte bleue.
22 Blanc, ///< Carte blanche.
23 Rouge, ///< Carte rouge.
24 Noir, ///< Carte noire.

30/**
31 * @brief Convertit une @ref CouleurCarte en chaîne de caractères lisible.
32 * @param couleur La couleur à convertir.
33 * @return Le nom de la couleur en français, ou @c "Inconnu" si non reconnue.
34 * @ingroup Cartes
35 */
36inline std::string couleurToString(CouleurCarte couleur)
37{
38 switch (couleur) {
39 case CouleurCarte::Bleu: return "Bleu";
40 case CouleurCarte::Blanc: return "Blanc";
41 case CouleurCarte::Rouge: return "Rouge";
42 case CouleurCarte::Noir: return "Noir";
43 case CouleurCarte::Jaune: return "Jaune";
44 case CouleurCarte::Vert: return "Vert";
45 case CouleurCarte::Locomotive: return "Locomotive";
46 default: return "Inconnu";
47 }
48}
