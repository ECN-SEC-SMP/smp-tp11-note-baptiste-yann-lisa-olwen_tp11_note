237/**
238 * @brief Vérifie si un ticket destination est complété par les routes du joueur.
239 *
240 * Construit un graphe à partir des routes appartenant au joueur, puis effectue
241 * un parcours en largeur (BFS) depuis la ville de départ du ticket pour
242 * déterminer si la ville d'arrivée est atteignable.
243 *
244 * @complexity O(V + E) où V est le nombre de villes et E le nombre de routes du joueur.
245 *
246 * @param t Ticket destination à vérifier.
247 * @param routes Liste complète des routes du plateau.
248 * @return @c true si le joueur relie les deux villes du ticket, @c false sinon.
249 */
250bool Joueur::verifierTicket(Ticket t, std::vector<Route>& routes)
251{
252 std::string villeA = t.getVilleA().getNom();
253 std::string villeB = t.getVilleB().getNom();
254
255 // Construire le graphe des routes du joueur
256 std::map<std::string, std::vector<std::string>> graphe;
257 for (Route& r : routes)
258 {
259 if (r.getProprio() == this) // seulement les routes du joueur
260 {
261 graphe[r.getVilleA().getNom()].push_back(r.getVilleB().getNom());
262 graphe[r.getVilleB().getNom()].push_back(r.getVilleA().getNom());
263 }
264 }
265
266 // BFS depuis villeA
267 std::vector<std::string> visites;
268 std::vector<std::string> file = {villeA};
269
270 while (!file.empty())
271 {
272 std::string ville = file.front();
273 file.erase(file.begin());
274
275 if (ville == villeB)
276 return true; // chemin trouvé !
277
278 // Marquer comme visitée
279 bool dejaVisite = false;
280 for (std::string v : visites)
281 if (v == ville) dejaVisite = true;
282
283 if (!dejaVisite)
284 {
285 visites.push_back(ville);
286 for (std::string voisin : graphe[ville])
287 file.push_back(voisin);
288 }
289 }
290 return false; // pas de chemin
291}
